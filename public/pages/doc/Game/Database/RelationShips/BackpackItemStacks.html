  self.update( amount: self.amount + add_amount )

def remove_amount(am)
  if am < self.amount      
    self.update( amount: self.amount - am )
    am = self.amount
  elsif am == self.amount
    am = self.amount
    self.remove()
  else
    raise "Invalid amount (too big)."
  end
  
  return am
end
