def self.setup()
  begin
    @data = JSON.parse( File.read( ENV_FILE ) )
    @data.each { |k,v| ENVS[k] = eval('"' + v + '"') }
  rescue Exception => e
    puts "Warning: Could not read envs.json file: " + e.message.to_s + "\n" + e.backtrace.to_s
  end
end
